//SETUP
type location;
type character : location; //character is built in but we give it more parents here - namely a location so if we want to add items people can have them (item at location)

property at (character : character) : place;

//ACTIONS
action travel(character : character, from : place, to : place) {
 precondition:
  from != to &
  (path(from, to) | path(to, from)) &
  alive(character) &
  at(character) == from;
 effect:
  at(character) = to;
 consenting: character;
 observing(c : character): alive(c) & (at(c) == from | at(c) == to);

};

//TRIGGERS
trigger see_at(character : character, other : character, place : place) {
 precondition:
  at(character) == place &
  at(other) == place &
  believes(character, at(other) != place);
 effect:
  believes(character, at(other) = place);
};
