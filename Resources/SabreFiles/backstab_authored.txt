//ACTIONS
action travel(character : character, from : place, to : place) {
 precondition:
  from != to &
  (path(from, to) | path(to, from)) &
  alive(character) &
  at(character) == from;
 effect:
  at(character) = to;
 consenting: character;
 observing(c : character): alive(c) & (at(c) == from | at(c) == to);

};

action kill(traitor : traitor, survivor : survivor, place : place) {
  precondition:
	at(traitor) == place &
	at(survivor) == place &
	alive(traitor) & alive(survivor);
  effect:
	!alive(survivor) &
	bad(traitor);
  consenting: traitor;
  observing(c : character): alive(c) & at(c) == place;
};

action complete(survivor : survivor, task : task) {
  precondition:
   alive(survivor) &
   !completed(survivor, task) &
   at(survivor) == at(task);
  effect:
    completed(survivor, task);
  consenting: survivor;
  observing(c: character): alive(c) & at(c) == at(survivor);
};

//TRIGGERS
trigger see_at(character : character, other : character, place : place) {
 precondition:
  alive(character) &
  at(character) == place &
  at(other) == place &
  believes(character, at(other) != place);
 effect:
  believes(character, at(other) = place);
};

trigger see_not_at(character : character, other : character, place : place) {
  precondition:
    alive(character) &
    at(character) == place &
    at (other) != place &
    believes(character, at(other) == place);
  effect:
    believes(character, at(other) = ?);
};

trigger see_alive(character : character, other : character) {
  precondition:
	alive(character) &
	at(character) == at(other) &
	alive(other) &
	believes(character, !alive(other));
  effect:
	believes(character, alive(other));
};

trigger see_dead(character : character, other : character) {
  precondition:
	alive(character) &
	at(character) == at(other) &
	!alive(other) &
	believes(character, alive(other));
  effect:
	believes(character, !alive(other));
};
